        def currency_sort(val):

            den = [100, 20, 10, 5, 1]
            val = int(val)

            scoop = []
            
            for i in range(0,len(den)):
                temp2 = int(temp/den[i])

                    scoop.append(f'{temp2} units of the denomination {den[i]}')
                temp = temp%den[i]
                    
            full_str = " | ".join([str(i) for i in scoop])
